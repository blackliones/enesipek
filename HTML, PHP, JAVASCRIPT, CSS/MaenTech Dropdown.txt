     <div class="mt-dropdown" id="mt-dropdown-category">
            <script>
                function DropdownButton(top_parent_id) {
                    const dropdown = document.getElementById(top_parent_id);
                    const window_id = dropdown.getElementsByClassName("mt-dropdown-window")[0].id;
                    const window = document.getElementById(window_id);
                    window.style.display = window.style.display == "block" ? "none" : "block"
                }
                function DropdownSelectedItem(item_id) {
                    const item = document.getElementById(item_id);//list item layer for example:-1
                    const window = document.getElementById(item.parentElement.id);//list window layer for example:0
                    const input_text = window.parentElement.getElementsByTagName("input")[0];//input text on the window layer for example:0
                    input_text.value = item.innerText;
                    window.style.display = "none";
                }
            </script>
            <div class="mt-dropdown-header" id="mt-dropdown-header"><!--MAENTECH DROPDOWN-->
                <input type="text" class="mt-dropdown-text" placeholder="Kategoriler">
                <div class="mt-dropdown-button" onclick="DropdownButton('mt-dropdown-category');">
                    <img tabindex="1" src="../Images/dropdown_button.png">
                </div>
                <on class="mt-dropdown-window" id="general-category-list">
                    <?php
                    foreach ($categories as $value) {
                        $id = $value['id'];
                        $text = $value['name'];
                        echo "<li onclick='DropdownSelectedItem($id)' name='$text' id='$id' tabindex='2'>$text</li>";
                    }
                    ?>
                </on>
            </div>
        </div>
